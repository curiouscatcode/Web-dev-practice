CREATE DATABASE expensedb;

CREATE TABLE users (
	id SERIAL NOT NULL PRIMARY KEY, 
	name TEXT NOT NULL,
	email TEXT NOT NULL,
	password TEXT NOT NULL
);

CREATE TABLE expense (
	id SERIAL NOT NULL PRIMARY KEY,
	user_id INT NOT NULL,
	title TEXT NOT NULL,
	amount FLOAT NOT NULL,
	category TEXT NOT NULL,
	date DATE NOT NULL,
	CONSTRAINT fk_expense FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
