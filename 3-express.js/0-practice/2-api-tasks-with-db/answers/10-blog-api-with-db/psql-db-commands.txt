CREATE TABLE blogs( 
	id SERIAL NOT NULL PRIMARY KEY, 
	title TEXT NOT NULL, 
	content TEXT NOT NULL,
	author TEXT NOT NULL,
	published_status BOOLEAN DEFAULT FALSE,
	created_status TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE comments (
	id SERIAL NOT NULL PRIMARY KEY,
	blog_id INT NOT NULL,
	user_name TEXT NOT NULL, 
	text TEXT NOT NULL,
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	CONSTRAINT fk_blog_comments FOREIGN KEY (blog_id) REFERENCES blogs(id) ON DELETE CASCADE
);

CREATE TABLE likes ( 
	id SERIAL NOT NULL PRIMARY KEY,
	blog_id INT NOT NULL,
	user_name TEXT NOT NULL,
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	CONSTRAINT fk_blog_likes FOREIGN KEY (blog_id) REFERENCES blogs(id) ON DELETE CASCADE
);
